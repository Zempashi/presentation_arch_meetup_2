\begin{frame}[fragile]

    \begin{block}{Specific tasks in playbooks.yml}

  tasks:
    - pacman:

        state: {absent|present|latest}
\end{minted}

\frametitle{Roles}
    \footnotesize{
    \begin{block}{Role depencies}
        \begin{minted}{text}
myplabook.yml
roles/
    myrole/
        tasks/main.yml  # playbook
        meta/main.yml
    aur_helper/
        tasks/main.yml
        \end{minted}
    \end{block}


    \begin{block}{roles/myrole/meta/main.yml}
        \begin{minted}{yaml}
dependencies:
  - role: aur_helper
    package_name: shinken
        \end{minted}
    \end{block}

    \begin{block}{myplaybook.yml}
        \begin{minted}{yaml}
- hosts: my_archlinux_server
  roles:
   - myrole
        \end{minted}
    \end{block}
    }
